import multer from 'multer';

const upload = multer({ 
  storage: multer.memoryStorage(),
  limits: { fileSize: 5 * 1024 * 1024 } // 5MB max
});

app.post('/api/parse-resume', upload.single('resume'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: 'No file uploaded' });
    }

    // Only accept PDFs for now
    if (req.file.mimetype !== 'application/pdf') {
      return res.status(400).json({ 
        error: 'Only PDF resumes are supported. Please convert your Word document to PDF first.' 
      });
    }

    const pdfBuffer = req.file.buffer;
    const extractedData = await parseResumeWithGemini(pdfBuffer);
    
    console.log('[Resume Parse] Successfully extracted data:', {
      name: extractedData.name,
      rolesCount: extractedData.roles?.length || 0
    });
    
    res.json(extractedData);
  } catch (error) {
    console.error('[Resume Parse] Error:', error);
    res.status(500).json({ 
      error: 'Failed to parse resume. Please try again or fill the form manually.' 
    });
  }
});
```

PART 2: ADD GEMINI PARSING FUNCTION

In server/ai-processor.ts:
```typescript
import { gemini } from './gemini'; // or wherever your Gemini client is

export async function parseResumeWithGemini(pdfBuffer: Buffer) {
  const base64Pdf = pdfBuffer.toString('base64');
  
  const prompt = `
You are a resume data extraction expert. Extract structured career information from this resume PDF.

CRITICAL INSTRUCTIONS:
- Extract ALL information exactly as written
- Preserve original wording for achievements (don't paraphrase)
- Use "Present" for current roles (not "Current" or "Now")
- Standardize all dates to "MMM YYYY" format (e.g., "Jan 2020", "Mar 2019")
- If a field is missing, use empty string ("") or empty array ([])
- Extract skills as individual items, not grouped phrases

REQUIRED OUTPUT FORMAT (JSON ONLY, NO MARKDOWN):
{
  "name": "string (full name from resume)",
  "email": "string (email address if present, else empty)",
  "phone": "string (phone number if present, else empty)",
  "location": "string (city, country format, e.g., 'Singapore, SG' or 'London, UK')",
  "linkedin": "string (LinkedIn URL if present, else empty)",
  "summary": "string (professional summary/objective if present, 2-3 sentences max)",
  "roles": [
    {
      "title": "string (exact job title from resume)",
      "company": "string (company name)",
      "location": "string (city, country where role was based)",
      "startDate": "string (MMM YYYY format)",
      "endDate": "string (MMM YYYY or 'Present')",
      "achievements": [
        "string (each bullet point from resume, preserved exactly)"
      ]
    }
  ],
  "skills": [
    "string (individual skill, tool, or methodology)"
  ],
  "education": [
    {
      "degree": "string (degree name, e.g., 'Bachelor of Science in Computer Science')",
      "institution": "string (university/school name)",
      "year": "string (graduation year YYYY or range YYYY-YYYY)"
    }
  ]
}

EXAMPLES OF GOOD EXTRACTION:

Input: "Jan 2020 - Current"
Output: "startDate": "Jan 2020", "endDate": "Present"

Input: "2019-2021"  
Output: "startDate": "Jan 2019", "endDate": "Dec 2021"

Input bullet: "Led team of 15 engineers across 3 time zones"
Output: "Led team of 15 engineers across 3 time zones" (exact copy)

Input skills section: "Python, React, AWS, Agile methodology"
Output: ["Python", "React", "AWS", "Agile methodology"]

Return ONLY valid JSON. No markdown code fences, no explanations, no preamble.
`;

  try {
    const result = await gemini.generateContent([
      { text: prompt },
      { 
        inlineData: { 
          data: base64Pdf, 
          mimeType: 'application/pdf' 
        } 
      }
    ]);

    const responseText = result.response.text();
    
    // Strip markdown code fences if Gemini adds them despite instructions
    const cleanJson = responseText
      .replace(/```json\n?/g, '')
      .replace(/```\n?/g, '')
      .trim();
    
    const parsed = JSON.parse(cleanJson);
    
    // Validate required fields
    if (!parsed.name) {
      throw new Error('Could not extract name from resume');
    }
    
    // Ensure arrays exist even if empty
    parsed.roles = parsed.roles || [];
    parsed.skills = parsed.skills || [];
    parsed.education = parsed.education || [];
    
    return parsed;
    
  } catch (error) {
    console.error('[Gemini Parse] Error:', error);
    throw new Error('Failed to extract data from resume. The file may be corrupted or unreadable.');
  }
}
```

Add upload handler:
```typescript
const handleResumeUpload = async (event: React.ChangeEvent<HTMLInputElement>) => {
  const file = event.target.files?.[0];
  if (!file) return;

  // Validate file type
  if (file.type !== 'application/pdf') {
    alert('Please upload a PDF file. If you have a Word document, save it as PDF first.');
    return;
  }

  // Validate file size (5MB)
  if (file.size > 5 * 1024 * 1024) {
    alert('File too large. Please upload a resume under 5MB.');
    return;
  }

  setResumeUploading(true);
  setResumeFileName(file.name);

  try {
    const formData = new FormData();
    formData.append('resume', file);

    const response = await fetch('/api/parse-resume', {
      method: 'POST',
      body: formData
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || 'Failed to parse resume');
    }

    const extractedData = await response.json();

    // Pre-fill form state with extracted data
    setFormData(prev => ({
      ...prev,
      step1: {
        fullName: extractedData.name || '',
        email: extractedData.email || '',
        phone: extractedData.phone || '',
        location: extractedData.location || ''
      },
      step2: {
        linkedinUrl: extractedData.linkedin || '',
        personalWebsite: prev.step2?.personalWebsite || '',
        otherLinks: prev.step2?.otherLinks || ''
      },
      step3: {
        professionalSummary: extractedData.summary || ''
      },
      step4: {
        careerHistory: extractedData.roles.map(role => ({
          title: role.title,
          company: role.company,
          location: role.location,
          startDate: role.startDate,
          endDate: role.endDate,
          achievements: role.achievements || []
        }))
      },
      step8: {
        technicalSkills: extractedData.skills.join(', ') || ''
      },
      step9: {
        education: extractedData.education.map(edu => ({
          degree: edu.degree,
          institution: edu.institution,
          year: edu.year
        }))
      }
    }));

    setResumeExtracted(true);
    
    // Show success and move to Step 1
    alert(`✓ Resume analyzed successfully!\n\nWe've pre-filled your information from ${file.name}. Please review each section and add your war stories in the upcoming steps.`);
    
    setCurrentStep(1);

  } catch (error: any) {
    console.error('Resume upload error:', error);
    alert(`Failed to parse resume: ${error.message}\n\nYou can continue filling the form manually.`);
  } finally {
    setResumeUploading(false);
  }
};
```
