PART 3: CREATE STRIPE CHECKOUT ENDPOINT

FILE: server/routes.ts

ADD new endpoint:
```typescript
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2024-11-20.acacia',
});

// POST /api/create-checkout-session
app.post('/api/create-checkout-session', async (req, res) => {
  try {
    const { profileId, tier } = req.body;
    
    if (!profileId || !tier) {
      return res.status(400).json({ error: 'Missing profileId or tier' });
    }
    
    // Verify profile exists and is unpaid
    const profile = await storage.getProfileById(profileId);
    if (!profile) {
      return res.status(404).json({ error: 'Profile not found' });
    }
    if (profile.paymentStatus === 'paid') {
      return res.status(400).json({ error: 'Already paid' });
    }
    
    // Define pricing
    const prices = {
      launch: {
        amount: 19900, // $199 in cents
        name: 'Launch - AI Twin Portfolio',
        description: 'One-time creation, 90-day hosting, downloadable',
      },
      evolve: {
        amount: 39900, // $399 in cents
        name: 'Evolve - AI Twin Portfolio + Editing',
        description: 'Custom domain, unlimited edits, 12-month hosting',
      },
      concierge: {
        amount: 119900, // $1,199 in cents
        name: 'Concierge - White-Glove AI Twin',
        description: 'Strategy interview, professional copywriting, launch posts',
      },
    };
    
    const selectedPrice = prices[tier as keyof typeof prices];
    if (!selectedPrice) {
      return res.status(400).json({ error: 'Invalid tier' });
    }
    
    // Create Stripe checkout session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      line_items: [
        {
          price_data: {
            currency: 'usd',
            product_data: {
              name: selectedPrice.name,
              description: selectedPrice.description,
            },
            unit_amount: selectedPrice.amount,
          },
          quantity: 1,
        },
      ],
      mode: 'payment',
      success_url: `${process.env.BASE_URL}/payment/success?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.BASE_URL}/payment/cancelled?profile_id=${profileId}`,
      metadata: {
        profileId,
        tier,
      },
    });
    
    // Update profile with session ID
    await storage.updateProfileById(profileId, {
      stripeSessionId: session.id,
      tier,
    });
    
    res.json({ sessionId: session.id, url: session.url });
  } catch (error) {
    console.error('[Stripe] Checkout error:', error);
    res.status(500).json({ error: 'Failed to create checkout session' });
  }
});
```

PART 4: CREATE STRIPE WEBHOOK ENDPOINT

ADD webhook handler (this confirms payment and makes portfolio public):
```typescript
import { buffer } from 'express';

// IMPORTANT: This route must parse raw body, not JSON
app.post('/api/stripe/webhook', 
  express.raw({ type: 'application/json' }),
  async (req, res) => {
    const sig = req.headers['stripe-signature'] as string;
    const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;
    
    let event: Stripe.Event;
    
    try {
      event = stripe.webhooks.constructEvent(req.body, sig, webhookSecret);
    } catch (err: any) {
      console.error('[Stripe] Webhook signature verification failed:', err.message);
      return res.status(400).send(`Webhook Error: ${err.message}`);
    }
    
    // Handle successful payment
    if (event.type === 'checkout.session.completed') {
      const session = event.data.object as Stripe.Checkout.Session;
      const { profileId, tier } = session.metadata || {};
      
      if (profileId) {
        try {
          // Update profile to "paid" and make PUBLIC
          await storage.updateProfileById(profileId, {
            paymentStatus: 'paid',
            paidAt: new Date(),
            stripePriceId: session.payment_intent as string,
            isPublic: true, // CRITICAL: Make portfolio public
          });
          
          console.log(`[Stripe] Payment confirmed for profile ${profileId}, tier: ${tier}. Portfolio is now PUBLIC.`);
          
          // TODO: Send confirmation email with domain link
          
        } catch (error) {
          console.error('[Stripe] Error updating profile after payment:', error);
        }
      }
    }
    
    res.json({ received: true });
  }
);
```

FILE: server/routes.ts

```typescript
// POST /api/payment/cancel
app.post('/api/payment/cancel', async (req, res) => {
  try {
    const { profileId } = req.body;
    
    if (!profileId) {
      return res.status(400).json({ error: 'Missing profileId' });
    }
    
    // Verify profile is unpaid
    const profile = await storage.getProfileById(profileId);
    
    if (!profile) {
      return res.status(404).json({ error: 'Profile not found' });
    }
    
    if (profile.paymentStatus === 'paid') {
      return res.status(400).json({ error: 'Cannot delete paid portfolio' });
    }
    
    // DELETE the unpaid portfolio
    await storage.deleteProfileById(profileId);
    
    console.log(`[Payment] Cancelled - deleted unpaid portfolio ${profileId}`);
    
    res.json({ success: true, message: 'Portfolio deleted' });
    
  } catch (error) {
    console.error('[Payment] Cancel error:', error);
    res.status(500).json({ error: 'Failed to cancel' });
  }
});
```

PART 7: PAYMENT CANCELLATION PAGE

FILE: Create client/src/pages/PaymentCancelled.tsx
```typescript
import { useEffect, useState } from 'react';
import { useSearchParams, useNavigate } from 'react-router-dom';

export default function PaymentCancelled() {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const [deleting, setDeleting] = useState(false);
  
  const profileId = searchParams.get('profile_id');
  
  const handleRetry = () => {
    // Go back to payment gate
    navigate(`/dashboard`);
  };
  
  const handleDelete = async () => {
    if (!profileId) return;
    
    setDeleting(true);
    try {
      await fetch('/api/payment/cancel', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ profileId }),
      });
      
      alert('Portfolio deleted. You can create a new one anytime.');
      navigate('/');
      
    } catch (error) {
      console.error('Delete error:', error);
      alert('Failed to delete. Please contact support.');
      setDeleting(false);
    }
  };
  
  return (
    <div className="min-h-screen bg-zinc-950 flex items-center justify-center p-6">
      <div className="max-w-md w-full text-center">
        <div className="text-6xl mb-4">⚠️</div>
        <h1 className="text-3xl font-bold mb-4">Payment Cancelled</h1>
        <p className="text-white/70 mb-8">
          Your portfolio is still in draft mode. It won't be published until payment is complete.
        </p>
        
        <div className="space-y-4">
          <button
            onClick={handleRetry}
            className="w-full bg-green-600 hover:bg-green-700 text-white font-bold py-3 rounded-lg transition-colors"
          >
            Retry Payment
          </button>
          
          <button
            onClick={handleDelete}
            disabled={deleting}
            className="w-full bg-zinc-800 hover:bg-zinc-700 text-white/70 font-bold py-3 rounded-lg transition-colors disabled:opacity-50"
          >
            {deleting ? 'Deleting...' : 'Delete Draft Portfolio'}
          </button>
        </div>
        
        <p className="text-xs text-white/50 mt-6">
          Draft portfolios are automatically deleted after 7 days if unpaid.
        </p>
      </div>
    </div>
  );
}
```

```typescript
import { useEffect, useState } from 'react';
import { useSearchParams, useNavigate } from 'react-router-dom';

export default function PaymentSuccess() {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const [loading, setLoading] = useState(true);
  const [domain, setDomain] = useState<string>('');
  const sessionId = searchParams.get('session_id');
  
  useEffect(() => {
    if (!sessionId) return;
    
    // Poll backend to check if payment is confirmed
    const checkPayment = setInterval(async () => {
      try {
        const res = await fetch(`/api/payment/status?session_id=${sessionId}`);
        const data = await res.json();
        
        if (data.paid) {
          setDomain(data.publicDomain);
          setLoading(false);
          clearInterval(checkPayment);
        }
      } catch (error) {
        console.error('Status check error:', error);
      }
    }, 2000); // Check every 2 seconds
    
    // Stop checking after 30 seconds
    setTimeout(() => {
      clearInterval(checkPayment);
      if (loading) {
        setLoading(false);
      }
    }, 30000);
    
    return () => clearInterval(checkPayment);
  }, [sessionId, loading]);
  
  if (loading) {
    return (
      <div className="min-h-screen bg-zinc-950 flex items-center justify-center">
        <div className="text-center">
          <div className="text-6xl mb-4 animate-pulse">⏳</div>
          <h1 className="text-2xl font-bold mb-2">Processing Payment...</h1>
          <p className="text-white/70">Confirming your transaction</p>

  return (
    <div className="min-h-screen bg-zinc-950 flex items-center justify-center p-6">

        <div className="text-6xl mb-6">🎉</div>

            href={`https://${domain}`}
            target="_blank"
            rel="noopener noreferrer"
            className="text-2xl font-mono text-green-500 hover:text-green-400 break-all"
          >
            {domain}
          </a>
          <button
            onClick={() => navigator.clipboard.writeText(`https://${domain}`)}
            className="mt-4 text-sm text-white/60 hover:text-white transition-colors"
          >
            📋 Copy Link
          </button>
        </div>
        
        <div className="space-y-3 text-left bg-white/5 border border-white/10 rounded-xl p-6 mb-8">
          <h3 className="font-bold text-lg mb-3">Next Steps:</h3>
          <div className="flex items-start gap-3">
```

```typescript
async getProfileByStripeSessionId(sessionId: string) {
  const result = await db
    .select()
    .from(twinProfiles)
```
