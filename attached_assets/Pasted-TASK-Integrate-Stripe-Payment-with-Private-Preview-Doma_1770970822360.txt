```typescript
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2024-11-20.acacia',

// POST /api/create-checkout-session
app.post('/api/create-checkout-session', async (req, res) => {
  try {
    const { profileId, tier } = req.body;
    
    if (!profileId || !tier) {
      return res.status(400).json({ error: 'Missing profileId or tier' });
    }
    
    // Verify profile exists and is unpaid
    const profile = await storage.getProfileById(profileId);
    if (!profile) {
      return res.status(404).json({ error: 'Profile not found' });
    }
    if (profile.paymentStatus === 'paid') {
      return res.status(400).json({ error: 'Already paid' });
    }
    
    // Define pricing with exact features
    const prices = {
      launch: {
        amount: 19900, // $199 in cents
        name: 'Launch - AI Twin Portfolio',
        description: 'AI portfolio + chatbot, 6-month hosting, downloadable',
      },
      evolve: {
        amount: 39900, // $399 in cents
        name: 'Evolve - AI Twin Portfolio + Editing',
        description: 'Custom domain, portfolio editor, tune chatbot, theme switcher, analytics, 12-month hosting',
      },
      concierge: {
        amount: 119900, // $1,199 in cents
        name: 'Concierge - White-Glove AI Twin',
        description: '90-min interview, professional copywriting, white-glove build, advanced tuning, priority support',
      },
    };
    
    const selectedPrice = prices[tier as keyof typeof prices];
    if (!selectedPrice) {
      return res.status(400).json({ error: 'Invalid tier' });
    }
    
    // Create Stripe checkout session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      line_items: [
        {
          price_data: {
            currency: 'usd',
            product_data: {
              name: selectedPrice.name,
              description: selectedPrice.description,
            },
            unit_amount: selectedPrice.amount,
          },
          quantity: 1,
        },
      ],
      mode: 'payment',
      success_url: `${process.env.BASE_URL}/payment/success?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.BASE_URL}/payment/cancelled?profile_id=${profileId}`,
      metadata: {
        profileId,
        tier,
      },
    });
    
    // Update profile with session ID
    await storage.updateProfileById(profileId, {
      stripeSessionId: session.id,
      tier,
    });
    
    res.json({ sessionId: session.id, url: session.url });
  } catch (error) {
    console.error('[Stripe] Checkout error:', error);
    res.status(500).json({ error: 'Failed to create checkout session' });
  }
});
```

PART 4: CREATE STRIPE WEBHOOK ENDPOINT

ADD webhook handler (this confirms payment and makes portfolio public):
```typescript
// IMPORTANT: This route must parse raw body, not JSON
app.post('/api/stripe/webhook', 
  express.raw({ type: 'application/json' }),
  async (req, res) => {
    const sig = req.headers['stripe-signature'] as string;
    const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;
    
    let event: Stripe.Event;
    
    try {
      event = stripe.webhooks.constructEvent(req.body, sig, webhookSecret);
    } catch (err: any) {
      console.error('[Stripe] Webhook signature verification failed:', err.message);
      return res.status(400).send(`Webhook Error: ${err.message}`);
    }
    
    // Handle successful payment
    if (event.type === 'checkout.session.completed') {
      const session = event.data.object as Stripe.Checkout.Session;
      const { profileId, tier } = session.metadata || {};
      
      if (profileId) {
        try {
          // Update profile to "paid" and make PUBLIC
          await storage.updateProfileById(profileId, {
            paymentStatus: 'paid',
            paidAt: new Date(),
            stripePriceId: session.payment_intent as string,
            isPublic: true, // CRITICAL: Make portfolio public
          });
          
          console.log(`[Stripe] Payment confirmed for profile ${profileId}, tier: ${tier}. Portfolio is now PUBLIC.`);
          
          // TODO: Send confirmation email with domain link
          
        } catch (error) {
          console.error('[Stripe] Error updating profile after payment:', error);
        }
      }
    }
    
    res.json({ received: true });
  }
);
```

```typescript
app.post('/api/payment/cancel', async (req, res) => {
  try {
    const { profileId } = req.body;
    
    if (!profileId) {
      return res.status(400).json({ error: 'Missing profileId' });
    }
    
    // Verify profile is unpaid
    const profile = await storage.getProfileById(profileId);
    
    if (!profile) {
      return res.status(404).json({ error: 'Profile not found' });
    }
    
    if (profile.paymentStatus === 'paid') {

  }
});
```

```typescript
import { useEffect, useState } from 'react';
import { useSearchParams, useNavigate } from 'react-router-dom';

export default function PaymentSuccess() {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const [loading, setLoading] = useState(true);
  const [domain, setDomain] = useState<string>('');
  const sessionId = searchParams.get('session_id');
  
  useEffect(() => {
    if (!sessionId) return;
    
    // Poll backend to check if payment is confirmed
    const checkPayment = setInterval(async () => {
      try {
        const res = await fetch(`/api/payment/status?session_id=${sessionId}`);
        const data = await res.json();
        
        if (data.paid) {
          setDomain(data.publicDomain);
          setLoading(false);
          clearInterval(checkPayment);
        }
      } catch (error) {
        console.error('Status check error:', error);
      }
    }, 2000); // Check every 2 seconds
    
    // Stop checking after 30 seconds
    setTimeout(() => {
      clearInterval(checkPayment);
      if (loading) {
        setLoading(false);
      }
    }, 30000);
    
    return () => clearInterval(checkPayment);
  }, [sessionId, loading]);
  
  if (loading) {
    return (
      <div className="min-h-screen bg-zinc-950 flex items-center justify-center">
        <div className="text-center">
```

```typescript
// GET /api/payment/status
app.get('/api/payment/status', async (req, res) => {
  try {
    const { session_id } = req.query;
    
    if (!session_id) {
      return res.status(400).json({ error: 'Missing session_id' });
    }
    
    // Find profile by session ID
    const profile = await storage.getProfileByStripeSessionId(session_id as string);
    
    if (!profile) {
      return res.status(404).json({ error: 'Profile not found' });
    }
    
    res.json({
      paid: profile.paymentStatus === 'paid',
      publicDomain: profile.publicDomain,
      tier: profile.tier,
    });
    
  } catch (error) {
    console.error('[Payment] Status check error:', error);
    res.status(500).json({ error: 'Server error' });
  }
});
```

```typescript
async getProfileByStripeSessionId(sessionId: string) {
  const result = await db
    .select()
    .from(twinProfiles)
    .where(eq(twinProfiles.stripeSessionId, sessionId))
    .limit(1);
  
  return result[0] || null;
}
```
